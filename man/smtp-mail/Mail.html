Class for compose and send mail thru a SMTP server. For example : 
let zz = new Mail.mail;;

zz#set_from "mail@mydomain";;

zz#set_rcpt "mail@yourdomain";;

zz#set_subject "important message";;

zz#set_server "smtp.mydomain";;

zz#send;;
